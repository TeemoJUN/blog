

## redis 锁

### 常常会有加错锁的时候。


```java
public class LockServiceImpl implements LockService {

    private final static Logger LOGGER = LoggerFactory.getLogger(LockServiceImpl.class);
    private static final String LOCK_PREFIX = "xxxxx_lock_prefix_";
    private static final String KEY = "xxxxxx_%s";

    @Resource
    private JedisClient jedisClient;

    private final static int lockSeconds = 30;

    @Override
    public boolean lock(String key) {
        return lock(key, lockSeconds);
    }

    @Override
    public boolean lock(String key, int seconds) {
        String lockKey = getLockKey(key);
        long timestamp;
        long now = System.currentTimeMillis();
        timestamp = now + seconds * 1000;
        try {
            long lock = jedisClient.setnx(lockKey, String.valueOf(timestamp));
            if (lock == 1 ||
                    (now > Long.valueOf(jedisClient.get(lockKey))
                            && now > Long.valueOf(jedisClient.getSet(lockKey, String.valueOf(timestamp))))) {
                //不想redis里累积太多key，尽管后续会被淘汰
                jedisClient.expire(lockKey, seconds * 2);
                return true;
            }
            return false;
        } catch (Exception e) {
            LOGGER.error("加锁失败[{}]", key, e);
            return false;
        }
    }

    @Override
    public void unlock(String key) {

        long now = System.currentTimeMillis();
        try {
            String lockKey = getLockKey(key);
            String redisValue = jedisClient.get(lockKey);
            if (StringUtils.isBlank(redisValue)) {
                return;
            }

            if (now < Long.valueOf(redisValue)) {
                jedisClient.del(lockKey);
            }
        } catch (Exception e) {
            LOGGER.error("解锁失败[{}]", key, e);
        }
    }

    @Override
    public boolean lockDevice(String deviceId, String userId, int expireSeconds) {
        String lockKey = buildKey(deviceId);
        try {
            long result = jedisClient.setnx(lockKey, userId);
            if (result == 1) {
                jedisClient.expire(lockKey, expireSeconds);
                return true;
            }else{
                return StringUtils.equals(jedisClient.get(lockKey), userId);
            }
        } catch (Exception e) {
            LOGGER.error("加锁失败[{}]", deviceId, e);
            unlockDevice(lockKey);
            return false;
        }
    }

    @Override
    public boolean unlockDevice(String deviceId) {
        try {
            String lockKey = buildKey(deviceId);
            jedisClient.del(lockKey);
            return true;
        } catch (Exception e) {
            LOGGER.error("解锁失败[{}]", deviceId, e);
        }
        return false;
    }

    private String getLockKey(String key) {
        return LOCK_PREFIX + key;
    }

    private String buildKey(String deviceId) {
        return String.format(KEY, deviceId);
    }
}
```
